23/**
24 * @brief ATB module initialization function.
25 * @details Configures timer to overflow with period of 10us to execute ATB_ISR.
26 */
27void ATB_Init(void)
28{
29 if(s_ATB_Initialized_U16 == (U16)0) /* Initialize only once. */
30 {
31 CpuTimer0Regs.TCR.bit.FREE = (U16)1; /* While debugging - do not stop the timer. */
32 CpuTimer0Regs.TPRH.all = (U16)0;
33 CpuTimer0Regs.TPR.bit.TDDR = (U16)19; /* Divide timer clock by 20. */
34 CpuTimer0Regs.PRD.all = (U32)99; /* Timer overflow every 10us. */
35 CpuTimer0Regs.TCR.bit.TIE = (U16)1; /* Enable timer overflow interrupt. */
36
37 /* Timer overflow interrupt setup. */
38 DINT; /* Disable interrupts. */
39 PieCtrlRegs.PIECTRL.bit.ENPIE = (U16)1; /* Enable interrupt vector table peripheral. */
40 PieCtrlRegs.PIEIER1.bit.INTx7 = (U16)1;
41
42 EALLOW;
43 PieVectTable.TIMER0_INT = &ATB_ISR; /* Write function pointer to the vector table. */
44 EDIS;
45 IER |= M_INT1; /* Enable CPU interrupt line. */
46 CpuTimer0Regs.TCR.bit.TSS = (U16)0; /* Start the system timer. */
47 EINT; /* Enable interrupts. */
48 s_ATB_Initialized_U16 = (U16)1;
49 }
50}

70/**
71 * @brief Check if requested ticks (time) have passed.
72 * @param ref_ticks_U32 Ticks reference for check.
73 * @param checked_ticks_passed_U32 Required number of ticks elapsed. To calculate this value from time you can use this macros:
74 * @arg @ref ATB_US_TO_TICKS_dM_U32()
75 * @arg @ref ATB_MS_TO_TICKS_dM_U32()
76 * @arg @ref ATM_S_TO_TICKS_dM_U32()
77 * @return Time passed status.
78 * @retval 1 number of ticks have passed
79 * @retval 0 number of ticks have not passed
80 *
81 * <b>Usage example:</b>
82 * @code{.c}
83 * U32 time1_U32 = ATB_GetTicks_U32();
84 * while(1)
85 * {
86 * if( ATB_CheckTicksPassed_U16(time1_U32, ATB_MS_TO_TICKS_dM_U32(250)) )
87 * {
88 * function() // Executed every 250ms.
89 * time1_U32 = ATB_GetTicks_U32(); // Write new reference time
90 * }
91 * }
92 * @endcode
93 */
94U16 ATB_CheckTicksPassed_U16(const uint32_t ref_ticks_U32, const uint32_t checked_ticks_passed_U32)
95{
96 const U32 ticks_timestamp_U32 = ATB_GetTicks_U32(); /* Saving current time stamp. */
97 U32 ticks_delta_U32 = ticks_timestamp_U32 - ref_ticks_U32; /* Calculation of delta of ticks. */
98 U16 status_U16 = (U16)0; /* Return value */
99
100 if(ticks_delta_U32 >= checked_ticks_passed_U32)
101 {
102 status_U16 = (U16)1;
103 }
104
105 return status_U16;
106}
