86struct SDRAM_CR_BITS { // bits description
87 Uint16 PAGESIGE:3; // 2:0 Page Size.

89 Uint16 IBANK:3; // 6:4 Internal Bank setup of SDRAM devices.
90 Uint16 rsvd2:1; // 7 Reserved

99 Uint16 rsvd8:1; // 19 Reserved
100 Uint16 rsvd9:3; // 22:20 Reserved
101 Uint16 rsvd10:3; // 25:23 Reserved
102 Uint16 rsvd11:3; // 28:26 Reserved
103 Uint16 PDWR:1; // 29 Perform refreshes during Power Down.
104 Uint16 PD:1; // 30 Power Down.
105 Uint16 SR:1; // 31 Self Refresh.

179struct SDRAM_TR_BITS { // bits description
180 Uint16 rsvd1:4; // 3:0 Reserved
